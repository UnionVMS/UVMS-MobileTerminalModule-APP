12  package eu.europa.ec.fisheries.uvms.mobileterminal.rest.dto;
13  
14  import java.util.ArrayList;
15  import java.util.HashMap;
16  import java.util.List;
17  import java.util.Map;
18  
19  import eu.europa.ec.fisheries.schema.mobileterminal.config.v1.Capability;
20  import eu.europa.ec.fisheries.schema.mobileterminal.config.v1.CapabilityConfiguration;
21  import eu.europa.ec.fisheries.schema.mobileterminal.config.v1.ConfigList;
22  import eu.europa.ec.fisheries.schema.mobileterminal.config.v1.TerminalSystemType;
23  
24  public class MobileTerminalConfig {
25  
26      private MobileTerminalConfig() {
27      }
28  
29      public static List<MobileTerminalDeviceConfig> mapConfigTransponders(List<TerminalSystemType> list) {
30          List<MobileTerminalDeviceConfig> transponders = new ArrayList<MobileTerminalDeviceConfig>();
31        
32          for (TerminalSystemType system : list) {
33  
34              MobileTerminalDeviceConfig transponder = new MobileTerminalDeviceConfig();
35              transponder.setTerminalSystemType(system.getType());
36  
37              if (system.getTerminalConfiguration() != null) {
38                  transponder.setTerminalFieldList(system.getTerminalConfiguration().getAttribute());
39              }
40              if (system.getComchannelConfiguration() != null) {
41                  transponder.setChannelFieldList(system.getComchannelConfiguration().getAttribute());
42              }
43  
44              if (system.getCapabilityConfiguration() != null) {
45                  CapabilityConfiguration configuration = system.getCapabilityConfiguration();
46                  if (configuration != null && configuration.getCapability() != null) {
47                      List<CapabilityDto> capabilityList = new ArrayList<>();
48                      for (Capability capability : configuration.getCapability()) {
49                          CapabilityDto dto = new CapabilityDto();
50                          dto.setName(capability.getName());
51                          if (capability.getOptions() != null) {
52                              dto.setOptionList(capability.getOptions());
53                          }
54                          capabilityList.add(dto);
55                      }
56                      transponder.setCapabilityList(capabilityList);
57                  }
58              }
59              transponders.add(transponder);
60          }
61          return transponders;
62      }
63  
64      public static Map<String, List<String>> mapConfigList(List<ConfigList> config) {
65          Map<String, List<String>> configValues = new HashMap<>();
66          for (ConfigList list : config) {
67              configValues.put(list.getName(), list.getValue());
68          }
69          return configValues;
70      }
71  }
