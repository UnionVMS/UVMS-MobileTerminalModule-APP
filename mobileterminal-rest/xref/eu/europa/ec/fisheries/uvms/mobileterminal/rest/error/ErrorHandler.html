26  /**
27   **/
28  public class ErrorHandler {
29      
30      public static ResponseDto getFault(Exception ex) {
31      	if(ex instanceof MobileTerminalServiceException) {
32      		if(ex instanceof InputArgumentException) {
33      			return new ResponseDto<String>(ex.getMessage(), ResponseCode.INPUT_ERROR);
34              }
35      		
36              if(ex instanceof MobileTerminalServiceMapperException) {
37              	return new ResponseDto<String>(ex.getMessage(), ResponseCode.MAPPING_ERROR);
38              }
39              return new ResponseDto<String>(ex.getMessage(), ResponseCode.SERVICE_ERROR);
40      	}
41      	
42      	if(ex instanceof MobileTerminalModelException) {
43      		if(ex instanceof MobileTerminalModelValidationException) {
44      			return new ResponseDto<String>(ex.getMessage(), ResponseCode.INPUT_ERROR);
45      		}
46      		
47      		if(ex instanceof MobileTerminalModelMapperException) {
48      			//MobileTerminalValidationException
49          		//MobileTerminalUnmarshallException
50      			return new ResponseDto<String>(ex.getMessage(), ResponseCode.MAPPING_ERROR);
51      		}
52      		
53      		if(ex instanceof MobileTerminalFaultException) {
54          		return extractFault((MobileTerminalFaultException)ex);
55          	}
56      		
57      		return new ResponseDto<String>(ex.getMessage(), ResponseCode.MODEL_ERROR);
58      	}
59      	
60      	if(ex instanceof MobileTerminalException) {
61      		return new ResponseDto<String>(ex.getMessage(), ResponseCode.MOBILE_TERMINAL_ERROR);
62      	}
63      	
64          return new ResponseDto<String>(ex.getMessage(), ResponseCode.UNDEFINED_ERROR);
65      }
66  
67      private static ResponseDto extractFault(MobileTerminalFaultException ex) {
68          MobileTerminalFault fault = ex.getMobileTerminalFault();
69          if (fault == null) {
70              return new ResponseDto<String>(ex.getMessage(), ResponseCode.DOMAIN_ERROR);
71          }
72  
73          MobileTerminalType terminal = fault.getTerminal();
74          if (terminal == null) {
75              return new ResponseDto<String>(fault.getMessage(), fault.getCode());
76          }
77  
78          return new ResponseDto<MobileTerminalType>(terminal, fault.getCode());
79      }
