12  package eu.europa.ec.fisheries.uvms.mobileterminal.rest.service;
13  
14  import java.util.List;
15  
16  import javax.ejb.EJB;
17  import javax.ejb.Stateless;
18  import javax.ws.rs.Consumes;
19  import javax.ws.rs.GET;
20  import javax.ws.rs.Path;
21  import javax.ws.rs.Produces;
22  import javax.ws.rs.core.MediaType;
23  
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  import eu.europa.ec.fisheries.schema.mobileterminal.config.v1.ConfigList;
28  import eu.europa.ec.fisheries.schema.mobileterminal.config.v1.TerminalSystemType;
29  import eu.europa.ec.fisheries.schema.mobileterminal.types.v1.SearchKey;
30  import eu.europa.ec.fisheries.uvms.mobileterminal.rest.dto.MobileTerminalConfig;
31  import eu.europa.ec.fisheries.uvms.mobileterminal.rest.dto.MobileTerminalDeviceConfig;
32  import eu.europa.ec.fisheries.uvms.mobileterminal.rest.dto.ResponseDto;
33  import eu.europa.ec.fisheries.uvms.mobileterminal.rest.error.ErrorHandler;
34  import eu.europa.ec.fisheries.uvms.mobileterminal.rest.error.ResponseCode;
35  import eu.europa.ec.fisheries.uvms.mobileterminal.service.ConfigService;
36  import eu.europa.ec.fisheries.uvms.rest.security.RequiresFeature;
37  import eu.europa.ec.fisheries.uvms.rest.security.UnionVMSFeature;
38  
39  /**
40   * @apiDescription Handles all Polls*/
41  @Path("/config")
42  @Stateless
43  @RequiresFeature(UnionVMSFeature.viewVesselsAndMobileTerminals)
44  public class ConfigResource {
45  
46      final static Logger LOG = LoggerFactory.getLogger(ConfigResource.class);
47  
48      @EJB
49      ConfigService configService;
50      
51      /**
52      *
53      * @responseMessage 200 Config transponders successfully retreived
54      * @responseMessage 500 Error when retrieveing the config values for
55      * transponders
56      *
57      * @summary Get all available config values for Mobile Terminals
58      *
59      */
60     @GET
61     @Consumes(value = {MediaType.APPLICATION_JSON})
62     @Produces(value = {MediaType.APPLICATION_JSON})
63     @Path("/transponders")
64  	public ResponseDto<List<MobileTerminalDeviceConfig>> getConfigTransponders() {
65         try {
66             LOG.info("Get config transponders invoked in rest layer.");
67             List<TerminalSystemType> list = configService.getTerminalSystems();
68             return new ResponseDto(MobileTerminalConfig.mapConfigTransponders(list), ResponseCode.OK);
69         } catch (Exception ex) {
70             LOG.error("[ Error when getting configTransponders ] {}", ex.getStackTrace());
71             return ErrorHandler.getFault(ex);
72         }
73     }
74  
75     /**
76      *
77      * @responseMessage 200 Config search fields successfully retreived
78      * @responseMessage 500 Error when retrieveing the config search fields for
79      * transponders
80      *
81      * @summary Gets all available search fields for Mobile Terminal
82      *
83      */
84     @GET
85     @Consumes(value = {MediaType.APPLICATION_JSON})
86     @Produces(value = {MediaType.APPLICATION_JSON})
87     @Path("/searchfields")
88     public ResponseDto<List<String>> getConfigSearchFields() {
89         LOG.info("Get config search fields invoked in rest layer.");
90         try {
91             return new ResponseDto(SearchKey.values(), ResponseCode.OK);
92         } catch (Exception ex) {
93             LOG.error("[ Error when getting config search fields ] {}", ex.getStackTrace());
94             return ErrorHandler.getFault(ex);
95         }
96     }
97      
98      @GET
99      @Consumes(value = {MediaType.APPLICATION_JSON})
100     @Produces(value = {MediaType.APPLICATION_JSON})
101     @Path("/")
102     public ResponseDto getConfiguration() {
103         try {
104         	List<ConfigList> config = configService.getConfig();
105             return new ResponseDto(MobileTerminalConfig.mapConfigList(config), ResponseCode.OK);
106         } catch (Exception ex) {
107             return ErrorHandler.getFault(ex);
108         }
109     }
110 }
