53      private final ResponseCode ERROR_RESULT;
54  
55      private final ResponseDto SUCCESS_RESULT_CREATE;
56      private final ResponseDto SUCCESS_RESULT_LIST_RESPONSE;
57      private final ResponseDto SUCCESS_RESULT_UPDATE;
58      private final ResponseDto SUCCESS_RESULT_GET_BY_ID;
59  
60      private final MobileTerminalType MOBILE_TERMINAL_DTO = MockData.createMobileTerminalDto(MOBILE_TERMINAL_ID_INT);
61      private final MobileTerminalListResponse MOBILE_TERMINAL_LIST_RESPONSE = MockData.createMobileTerminalListResponse();
62  
63      public ResponseTest() {
64          ERROR_RESULT = ResponseCode.UNDEFINED_ERROR;
65          SUCCESS_RESULT_UPDATE = new ResponseDto(MOBILE_TERMINAL_DTO, ResponseCode.OK);
66          SUCCESS_RESULT_LIST_RESPONSE = new ResponseDto(MOBILE_TERMINAL_LIST_RESPONSE, ResponseCode.OK);
67          SUCCESS_RESULT_CREATE = new ResponseDto(MOBILE_TERMINAL_DTO, ResponseCode.OK);
68          SUCCESS_RESULT_GET_BY_ID = new ResponseDto(MOBILE_TERMINAL_DTO, ResponseCode.OK);
69      }

71      @Before
72      public void setUp() {
73          MockitoAnnotations.initMocks(this);
74      }

76      @Test
77      public void testGetMobileTerminalById() throws MobileTerminalException {
78          doReturn(MOBILE_TERMINAL_DTO).when(mobileTerminalService).getMobileTerminalById(MOBILE_TERMINAL_ID);
79          ResponseDto result = mobileTerminalResource.getMobileTerminalById(MOBILE_TERMINAL_ID);
80          Mockito.verify(mobileTerminalService).getMobileTerminalById(MOBILE_TERMINAL_ID);
81          assertEquals(SUCCESS_RESULT_GET_BY_ID.toString(), result.toString());
82      }
83  
84      @Test
85      public void testGetMobileTerminalList() throws MobileTerminalException {
86          doReturn(MOBILE_TERMINAL_LIST_RESPONSE).when(mobileTerminalService).getMobileTerminalList(null);
87          ResponseDto result = mobileTerminalResource.getMobileTerminalList(null);
88          Mockito.verify(mobileTerminalService).getMobileTerminalList(null);
89          assertEquals(SUCCESS_RESULT_LIST_RESPONSE.toString(), result.toString());
90      }
91  
92      @Test
93      public void testUpdateMobileTeriminal() throws MobileTerminalException {
94          doReturn(MOBILE_TERMINAL_DTO).when(mobileTerminalService).updateMobileTerminal(MOBILE_TERMINAL_DTO, "", MobileTerminalSource.INTERNAL, "TEST");
95          doReturn("TEST").when(request).getRemoteUser();
96          ResponseDto result = mobileTerminalResource.updateMobileTerminal("", MOBILE_TERMINAL_DTO);
97          Mockito.verify(mobileTerminalService).updateMobileTerminal(MOBILE_TERMINAL_DTO, "", MobileTerminalSource.INTERNAL, "TEST");
98          assertEquals(SUCCESS_RESULT_UPDATE.toString(), result.toString());
99      }

103         doReturn(MOBILE_TERMINAL_DTO).when(mobileTerminalService).createMobileTerminal(MOBILE_TERMINAL_DTO, MobileTerminalSource.INTERNAL, "TEST");
104         doReturn("TEST").when(request).getRemoteUser();
105         ResponseDto result = mobileTerminalResource.createMobileTerminal(MOBILE_TERMINAL_DTO);
106         Mockito.verify(mobileTerminalService).createMobileTerminal(MOBILE_TERMINAL_DTO, MobileTerminalSource.INTERNAL, "TEST");
107         assertEquals(SUCCESS_RESULT_UPDATE.toString(), result.toString());
108     }

110     @Test
111     public void checkDtoReturnsValid() {
112 
113         String VALUE = "HELLO_DTO";
114         ResponseDto dto = new ResponseDto(VALUE, ResponseCode.OK);
115         Assert.assertEquals(dto.getCode().intValue(), ResponseCode.OK.getCode());
116         Assert.assertEquals(dto.getData(), VALUE);
117 
118         dto = new ResponseDto(null, ResponseCode.UNDEFINED_ERROR);
119         Assert.assertEquals(dto.getCode().intValue(), ResponseCode.UNDEFINED_ERROR.getCode());
120         Assert.assertEquals(dto.getData(), null);
121 
122     }
