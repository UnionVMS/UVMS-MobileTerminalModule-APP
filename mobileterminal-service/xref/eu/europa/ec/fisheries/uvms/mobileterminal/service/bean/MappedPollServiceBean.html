19  
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  
23  import eu.europa.ec.fisheries.schema.mobileterminal.polltypes.v1.PollListQuery;
24  import eu.europa.ec.fisheries.schema.mobileterminal.polltypes.v1.PollListResponse;
25  import eu.europa.ec.fisheries.schema.mobileterminal.polltypes.v1.PollRequestType;
26  import eu.europa.ec.fisheries.schema.mobileterminal.polltypes.v1.PollResponseType;
27  import eu.europa.ec.fisheries.schema.mobileterminal.polltypes.v1.PollableQuery;
28  import eu.europa.ec.fisheries.schema.mobileterminal.source.v1.MobileTerminalListResponse;
29  import eu.europa.ec.fisheries.schema.mobileterminal.types.v1.MobileTerminalType;

31  import eu.europa.ec.fisheries.uvms.mobileterminal.dto.PollChannelDto;

35  import eu.europa.ec.fisheries.uvms.mobileterminal.model.exception.MobileTerminalException;
36  import eu.europa.ec.fisheries.uvms.mobileterminal.service.MappedPollService;
37  import eu.europa.ec.fisheries.uvms.mobileterminal.service.MobileTerminalService;
38  import eu.europa.ec.fisheries.uvms.mobileterminal.service.PollService;
39  import eu.europa.ec.fisheries.uvms.mobileterminal.service.exception.MobileTerminalServiceException;
40  
41  @Stateless
42  public class MappedPollServiceBean implements MappedPollService {
43      final static Logger LOG = LoggerFactory.getLogger(MappedPollServiceBean.class);
44  
45      @EJB
46      PollService pollService;
47  
48      @EJB
49      MobileTerminalService mobileTerminalService;
50  
51      @Override
52      public CreatePollResultDto createPoll(PollRequestType pollRequest, String username) throws MobileTerminalServiceException {
53          LOG.debug("Create poll");
54  
55          CreatePollResultDto pollResponse = pollService.createPoll(pollRequest, username);
56          return pollResponse;
57      }
58  
59      @Override
60      public List<PollDto> getRunningProgramPolls() throws MobileTerminalServiceException {
61          LOG.debug("Get running program polls");
62  
63          List<PollResponseType> pollResponse = pollService.getRunningProgramPolls();
64          return PollMapper.mapPolls(pollResponse);
65      }
66  
67      @Override
68      public PollDto startProgramPoll(String pollId, String username) throws MobileTerminalServiceException {
69          PollResponseType pollResponse = pollService.startProgramPoll(pollId, username);
70          return PollMapper.mapPoll(pollResponse);
71      }
72  
73      @Override
74      public PollDto stopProgramPoll(String pollId, String username) throws MobileTerminalServiceException {
75          PollResponseType pollResponse = pollService.stopProgramPoll(pollId, username);
76          return PollMapper.mapPoll(pollResponse);
77      }
78  
79      @Override
80      public PollDto inactivateProgramPoll(String pollId, String username) throws MobileTerminalServiceException {
81          PollResponseType pollResponse = pollService.inactivateProgramPoll(pollId, username);
82          return PollMapper.mapPoll(pollResponse);
83      }
84  
85      @Override
86      public PollChannelListDto getPollBySearchQuery(PollListQuery query) throws MobileTerminalServiceException {
87      	PollChannelListDto channelListDto = new PollChannelListDto();
88      	
89      	PollListResponse pollResponse = pollService.getPollBySearchCriteria(query);
90          channelListDto.setCurrentPage(pollResponse.getCurrentPage());
91          channelListDto.setTotalNumberOfPages(pollResponse.getTotalNumberOfPages());
92      	
93          ArrayList<PollChannelDto> pollChannelList = new ArrayList<>();
94          for(PollResponseType responseType : pollResponse.getPollList()) {
95          	PollChannelDto terminal = PollMapper.mapPollChannel(responseType.getMobileTerminal());
96          	terminal.setPoll(PollMapper.mapPoll(responseType));
97          	pollChannelList.add(terminal);
98          }
99          channelListDto.setPollableChannels(pollChannelList);
100         return channelListDto;
101     }
102 
103     @Override
104     public PollChannelListDto getPollableChannels(PollableQuery query) throws MobileTerminalException {
105         PollChannelListDto channelListDto = new PollChannelListDto();
106 
107         MobileTerminalListResponse response = mobileTerminalService.getPollableMobileTerminal(query);
108         channelListDto.setCurrentPage(response.getCurrentPage());
109         channelListDto.setTotalNumberOfPages(response.getTotalNumberOfPages());
110 
111         ArrayList<PollChannelDto> pollChannelList = new ArrayList<>();
112         for(MobileTerminalType terminalType : response.getMobileTerminal()) {
113         	PollChannelDto terminal = PollMapper.mapPollChannel(terminalType);
114         	pollChannelList.add(terminal);
115         }
116         channelListDto.setPollableChannels(pollChannelList);
117         return channelListDto;
118     }
119 }
