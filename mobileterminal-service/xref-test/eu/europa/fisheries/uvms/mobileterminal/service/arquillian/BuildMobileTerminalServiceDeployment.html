19  @ArquillianSuiteDeployment
20  public abstract class BuildMobileTerminalServiceDeployment {
21  
22      @Deployment(name = "normal", order = 1)
23      public static Archive<?> createDeployment() {
24  
25          // Import Maven runtime dependencies
26          File[] files = Maven.resolver().loadPomFromFile("pom.xml")
27                  .importRuntimeDependencies().resolve().withTransitivity().asFile();
28  
29          // Embedding war package which contains the test class is needed
30          // So that Arquillian can invoke test class through its servlet test runner
31          WebArchive testWar = ShrinkWrap.create(WebArchive.class, "test.war");
32          testWar.addPackages(true, "com.tocea.easycoverage.framework.api");
33          testWar.addPackages(true, "eu.europa.fisheries.uvms.mobileterminal.service");
34  
35          testWar.addPackages(true, "eu.europa.ec.fisheries.schema");
36          testWar.addClass(TransactionalTests.class);
37          testWar.addClass(ConfigService.class);
38          testWar.addClass(ConfigServiceBean.class);
39          testWar.addClass(ConfigModel.class);
40          testWar.addClass(TransactionalTests.class);
41          testWar.addClass(MobileTerminalMessageException.class);
42          testWar.addClass(MobileTerminalConfigHelper.class);
43  
44          // Empty beans for EE6 CDI
45          testWar.addAsWebInfResource(EmptyAsset.INSTANCE, "beans.xml");
46          testWar.addAsLibraries(files);
47  
48  
49  
50          return testWar;
51      }
52
