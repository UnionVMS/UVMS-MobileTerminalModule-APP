15  import java.util.List;

17  import org.slf4j.Logger;
18  import org.slf4j.LoggerFactory;

20  import eu.europa.ec.fisheries.schema.mobileterminal.polltypes.v1.PollAttribute;
21  import eu.europa.ec.fisheries.schema.mobileterminal.polltypes.v1.PollAttributeType;
22  import eu.europa.ec.fisheries.schema.mobileterminal.polltypes.v1.PollId;

28  import eu.europa.ec.fisheries.uvms.mobileterminal.entity.poll.PollBase;
29  import eu.europa.ec.fisheries.uvms.mobileterminal.entity.poll.PollProgram;

32  
33  public class PollEntityToModelMapper {
34      private static Logger LOG = LoggerFactory.getLogger(PollEntityToModelMapper.class);
35  
36      private static PollResponseType mapToPollResponseType(PollBase pollBase, MobileTerminalType mobileTerminalType) throws MobileTerminalModelMapperException {
37          PollResponseType response = new PollResponseType();
38          response.setComment(pollBase.getComment());
39          response.setUserName(pollBase.getUser());
40          // TODO created time?
41          // response.setCreatedTime();
42          response.setMobileTerminal(mobileTerminalType);
43          response.getAttributes().add(createPollAttribute(PollAttributeType.USER, pollBase.getUser()));
44          return response;
45      }
46  
47      public static PollResponseType mapToPollResponseType(PollProgram program, MobileTerminalType mobileTerminalType) throws MobileTerminalModelMapperException {
48          PollResponseType response = mapToPollResponseType(program.getPollBase(), mobileTerminalType);
49          response.setPollType(PollType.PROGRAM_POLL);
50          PollId pollId = new PollId();
51          pollId.setGuid(program.getGuid());
52          response.setPollId(pollId);
53  
54          response.getAttributes().addAll(getProgramPollAttributes(program));
55          return response;
56      }
57  
58      public static PollResponseType mapToPollResponseType(Poll poll, MobileTerminalType mobileTerminalType, PollType pollType) throws MobileTerminalModelMapperException {
59          PollResponseType response = mapToPollResponseType(poll.getPollBase(), mobileTerminalType);
60          response.setPollType(pollType);
61          PollId pollId = new PollId();
62          pollId.setGuid(poll.getGuid());
63          response.setPollId(pollId);
64          
65          return response;
66      }
67  
68      private static List<PollAttribute> getProgramPollAttributes(PollProgram program) {
69          List<PollAttribute> attributes = new ArrayList<PollAttribute>();
70          attributes.add(createPollAttribute(PollAttributeType.FREQUENCY, program.getFrequency().toString()));
71          attributes.add(createPollAttribute(PollAttributeType.START_DATE, DateUtils.parseUTCDateTimeToString(program.getStartDate())));
72          attributes.add(createPollAttribute(PollAttributeType.END_DATE, DateUtils.parseUTCDateTimeToString(program.getStopDate())));
73  
74          switch (program.getPollState()) {
75          case STARTED:
76          	attributes.add(createPollAttribute(PollAttributeType.PROGRAM_RUNNING, MobileTerminalConstants.TRUE));
77              break;
78          case STOPPED:
79          case ARCHIVED:
80          	attributes.add(createPollAttribute(PollAttributeType.PROGRAM_RUNNING, MobileTerminalConstants.FALSE));
81              break;
82          }
83          return attributes;
84      }
85  
86      private static PollAttribute createPollAttribute(PollAttributeType key, String value) {
87          PollAttribute attrib = new PollAttribute();
88          attrib.setKey(key);
89          attrib.setValue(value);
90          return attrib;
91      }
92  }
