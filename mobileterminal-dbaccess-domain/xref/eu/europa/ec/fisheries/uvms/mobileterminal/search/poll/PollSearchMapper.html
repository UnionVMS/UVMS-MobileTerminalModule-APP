25  public class PollSearchMapper {
26  
27  	public static List<PollSearchKeyValue> createSearchFields(List<ListCriteria> criterias) throws SearchMapperException {
28  		Map<PollSearchField, PollSearchKeyValue> searchKeyValues = new HashMap<>();
29          for (ListCriteria criteria : criterias) {
30          	PollSearchKeyValue keyValue = mapSearchKey(criteria, searchKeyValues);
31              searchKeyValues.put(keyValue.getSearchField(), keyValue);
32          }
33          return new ArrayList<PollSearchKeyValue>(searchKeyValues.values());
34  	}
35  	
36      private static PollSearchKeyValue mapSearchKey(ListCriteria criteria, Map<PollSearchField, PollSearchKeyValue> searchKeys) throws SearchMapperException {
37          if (criteria == null || criteria.getKey() == null || criteria.getValue() == null) {
38              throw new SearchMapperException("Non valid search criteria");
39          }
40  
41          PollSearchKeyValue searchKeyValue = getSearchKeyValue(getSearchField(criteria.getKey()), searchKeys);
42          searchKeyValue.getValues().add(criteria.getValue());
43          return searchKeyValue;
44      }
45  	
46      private static PollSearchKeyValue getSearchKeyValue(PollSearchField field, Map<PollSearchField, PollSearchKeyValue> searchKeys) {
47      	PollSearchKeyValue searchKeyValue = searchKeys.get(field);
48          if (searchKeyValue == null) {
49              searchKeyValue = new PollSearchKeyValue();
50          }
51          searchKeyValue.setSearchField(field);
52          return searchKeyValue;
53      }
54      
55  	private static PollSearchField getSearchField(SearchKey key) throws SearchMapperException {
56  		switch(key) {
57  		case CONNECT_ID:
58  			return PollSearchField.CONNECT_ID;
59  		case POLL_ID:
60  			return PollSearchField.POLL_ID;
61  		case POLL_TYPE:
62  			return PollSearchField.POLL_TYPE;
63  		case TERMINAL_TYPE:
64  			return PollSearchField.TERMINAL_TYPE;
65  		case USER:
66  			return PollSearchField.USER;
67  		default:
68  			throw new SearchMapperException("No searchKey " + key.name());
69  		}
70  	}
71  
72  	private static String createSearchSql(List<PollSearchKeyValue> searchKeys, boolean isDynamic) {
73  		StringBuilder builder = new StringBuilder();
74  		String OPERATOR = " OR ";
75  		if(isDynamic) {
76  			OPERATOR = " AND ";
77  		}
78  		
79  		builder.append(" INNER JOIN p.pollBase pb ");
80  		builder.append(" INNER JOIN pb.mobileterminal mt ");
81  		
82  		if(!searchKeys.isEmpty()) {
83  			builder.append(" WHERE ");
84  			boolean first = true;
85  			for(PollSearchKeyValue keyValue : searchKeys) {
86  				if(first) {
87  					first = false;
88  				} else {
89  					builder.append(OPERATOR);
90  				}
91  				builder.append(keyValue.getSearchField().getTable().getTableAlias()).append(".").append(keyValue.getSearchField().getSqlColumnName());
92  				builder.append(" IN (:").append(keyValue.getSearchField().getSqlReplaceToken()).append(") ");
93  			}
94  		}
95  		
96  		return builder.toString();
97  	}
98  	
99  	public static String createCountSearchSql(List<PollSearchKeyValue> searchKeys, boolean isDynamic) {
100 		StringBuilder builder = new StringBuilder();
101 		builder.append("SELECT COUNT (DISTINCT p) FROM Poll p ");
102 		builder.append(createSearchSql(searchKeys, isDynamic));
103 		return builder.toString();
104 	}
105 
106 	public static String createSelectSearchSql(List<PollSearchKeyValue> searchKeys, boolean isDynamic) {
107 		StringBuilder builder = new StringBuilder();
108 		builder.append("SELECT DISTINCT p FROM Poll p ");
109 		builder.append(createSearchSql(searchKeys, isDynamic));
110 		return builder.toString();
111 	}
112 	
113 	public static String createPollableSearchSql(List<String> idList) {
114 		StringBuilder builder = new StringBuilder();
115 		builder.append("SELECT DISTINCT c FROM Channel c");
116 		builder.append(" INNER JOIN FETCH c.mobileTerminal mt");
117 		builder.append(" INNER JOIN FETCH mt.mobileTerminalEvents me");
118 		builder.append(" INNER JOIN FETCH me.pollChannel pc");
119 		builder.append(" INNER JOIN FETCH mt.plugin p ");
120 		builder.append(" INNER JOIN FETCH p.capabilities cap ");
121 		builder.append(" WHERE ");
122 		builder.append(" c.guid = pc.guid ");
123 		builder.append(" AND me.active = true ");
124 		builder.append(" AND mt.archived = '0' AND mt.inactivated = '0' AND p.pluginInactive = '0' ");
125 		builder.append(" AND (cap.name = 'POLLABLE' AND UPPER(cap.value) = 'TRUE' ) ");
126 		builder.append(" AND (me.connectId is not null) ");
127 		if( idList != null && !idList.isEmpty()) {
128 			builder.append(" AND me.connectId IN :idList");
129 		}
130 		builder.append(" ORDER BY c.guid DESC ");
131 		return builder.toString();
132 	}
