12  package eu.europa.ec.fisheries.uvms.mobileterminal.dao.bean;
13  
14  import java.util.List;
15  
16  import javax.ejb.Stateless;
17  import javax.persistence.NoResultException;
18  import javax.persistence.TypedQuery;
19  
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  
23  import eu.europa.ec.fisheries.uvms.mobileterminal.constant.MobileTerminalConstants;
24  import eu.europa.ec.fisheries.uvms.mobileterminal.dao.Dao;
25  import eu.europa.ec.fisheries.uvms.mobileterminal.dao.OceanRegionDao;
26  import eu.europa.ec.fisheries.uvms.mobileterminal.dao.exception.ConfigDaoException;
27  import eu.europa.ec.fisheries.uvms.mobileterminal.entity.OceanRegion;
28  
29  @Stateless
30  public class OceanRegionDaoBean extends Dao implements OceanRegionDao {
31  	final static Logger LOG = LoggerFactory.getLogger(OceanRegionDaoBean.class);
32  
33  	@Override
34  	public List<OceanRegion> getOceanRegionList() throws ConfigDaoException {
35  		try {
36              TypedQuery<OceanRegion> query = em.createNamedQuery(MobileTerminalConstants.OCEAN_REGIONS, OceanRegion.class);
37              return query.getResultList();
38          } catch (NoResultException e) {
39              LOG.error("[ Error when getting ocean regions. ] {}", e.getMessage());
40              throw new ConfigDaoException("No entities found when retrieving ocean regions");
41          }
42  	}
43  	
44  }
