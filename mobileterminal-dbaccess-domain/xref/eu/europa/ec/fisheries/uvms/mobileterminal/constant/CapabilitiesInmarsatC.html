13  package eu.europa.ec.fisheries.uvms.mobileterminal.constant;
14  
15  import eu.europa.ec.fisheries.uvms.mobileterminal.entity.types.TerminalCapability;
16  
17  public class CapabilitiesInmarsatC {
18  
19      public static boolean getCapability(TerminalCapability capability) {
20          if(capability != null) {
21              switch(capability) {
22                  case SUPPORT_MULTIPLE_CHANNEL:
23                      return true;
24                  case PLUGIN:
25                      return true;
26                  case IS_CONFIGURABLE:
27                      return true;
28                  case IS_POLLABLE:
29                      return true;
30                  case SUPPORT_MULTIPLE_OCEAN:
31                      return true;
32                  case SUPPORT_SAMPLING:
33                      return true;
34                  case SUPPORT_SINGLE_OCEAN:
35                      return false;
36                  default:
37                      return false;
38              }
39          }
40          return false;
41      }
42  }
