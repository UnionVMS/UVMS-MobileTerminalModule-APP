12  package eu.europa.ec.fisheries.uvms.mobileterminal.message.producer.bean;
13  
14  import javax.annotation.PostConstruct;
15  import javax.ejb.Stateless;
16  import javax.jms.Connection;
17  import javax.jms.ConnectionFactory;
18  import javax.jms.JMSException;
19  import javax.jms.QueueConnectionFactory;
20  import javax.jms.Session;
21  import javax.jms.TextMessage;
22  import javax.naming.InitialContext;
23  import javax.naming.NamingException;
24  
25  import org.slf4j.LoggerFactory;
26  
27  import eu.europa.ec.fisheries.uvms.mobileterminal.message.constants.MessageConstants;
28  
29  @Stateless
30  public class JMSConnectorBean {
31      final static org.slf4j.Logger LOG = LoggerFactory.getLogger(JMSConnectorBean.class);
32  
33      private ConnectionFactory connectionFactory;
34  
35      private Connection connection;
36  
37      @PostConstruct
38      private void connectToQueue() {
39          LOG.debug("Open connection to JMS broker");
40          InitialContext ctx;
41          try {
42              ctx = new InitialContext();
43          } catch (Exception e) {
44              LOG.error("Failed to get InitialContext",e);
45              throw new RuntimeException(e);
46          }
47          try {
48              connectionFactory = (QueueConnectionFactory) ctx.lookup(MessageConstants.CONNECTION_FACTORY);
49          } catch (NamingException ne) {
50              //if we did not find the connection factory we might need to add java:/ at the start
51              LOG.debug("Connection Factory lookup failed for " + MessageConstants.CONNECTION_FACTORY);
52              String wfName = "java:/" + MessageConstants.CONNECTION_FACTORY;
53              try {
54                  LOG.debug("trying " + wfName);
55                  connectionFactory = (QueueConnectionFactory) ctx.lookup(wfName);
56              } catch (Exception e) {
57                  LOG.error("Connection Factory lookup failed for both " + MessageConstants.CONNECTION_FACTORY  + " and " + wfName);
58                  throw new RuntimeException(e);
59              }
60          }
61          try {
62              connection = connectionFactory.createConnection();
63              connection.start();
64          } catch (JMSException ex) {
65              LOG.error("Error when open connection to JMS broker");
66          }
67      }
68  
69      public Session getNewSession() throws JMSException {
70          if (connection == null) {
71              connectToQueue();
72          }
73          Session session = connection.createSession(true, Session.AUTO_ACKNOWLEDGE);
74          return session;
75      }
76  
77      public TextMessage createTextMessage(Session session, String message) throws JMSException {
78          return session.createTextMessage(message);
79      }
80  }
