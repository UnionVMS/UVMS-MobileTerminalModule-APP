62  	@Test
63  	public void testGetPluginListEmpty() throws ConfigDaoException {
64  		List<MobileTerminalPlugin> result = new ArrayList<>();
65  		when(em.createNamedQuery(MobileTerminalConstants.PLUGIN_FIND_ALL, MobileTerminalPlugin.class)).thenReturn(query);
66  		when(query.getResultList()).thenReturn(result);
67  		
68  		List<MobileTerminalPlugin> pluginList = testDaoBean.getPluginList();
69  		
70  		assertEquals(0, pluginList.size());
71  	}

74  	public void testGetPluginListOne() throws ConfigDaoException {

79  		
80  		List<MobileTerminalPlugin> pluginList = testDaoBean.getPluginList();

129 	@Test
130 	public void testGetPluginByServiceName() throws NoEntityFoundException {
131 		String serviceClassName = "ServiceClassName";
132 		when(plugin.getName()).thenReturn(serviceClassName);
133 		when(em.createNamedQuery(MobileTerminalConstants.PLUGIN_FIND_BY_SERVICE_NAME, MobileTerminalPlugin.class)).thenReturn(query);
134 		when(query.getSingleResult()).thenReturn(plugin);
135 		
136 		MobileTerminalPlugin result = testDaoBean.getPluginByServiceName(serviceClassName);
137 		
138 		assertNotNull(result);
139 		assertEquals(serviceClassName, result.getName());
140 	}
141 	
142 	@Test(expected=NoEntityFoundException.class)
143 	public void testGetPluginByServiceNameException() throws NoEntityFoundException {
144 		String serviceClassName = "ServiceClassName";
145 		when(em.createNamedQuery(MobileTerminalConstants.PLUGIN_FIND_BY_SERVICE_NAME, MobileTerminalPlugin.class)).thenReturn(query);
146 		when(query.getSingleResult()).thenThrow(new NoResultException());
147 		
148 		MobileTerminalPlugin result = testDaoBean.getPluginByServiceName(serviceClassName);
149 	}
