12  package eu.europa.ec.fisheries.uvms.mobileterminal.constant;
13  
14  import static org.junit.Assert.assertFalse;
15  import static org.junit.Assert.assertTrue;
16  
17  import java.util.Arrays;
18  import java.util.HashSet;
19  
20  import org.junit.Test;
21  import org.junit.runner.RunWith;
22  import org.mockito.runners.MockitoJUnitRunner;
23  
24  import eu.europa.ec.fisheries.uvms.mobileterminal.entity.inmarsatc.InmarsatCHistoryOceanRegion;
25  
26  @RunWith(MockitoJUnitRunner.class)
27  public class TestEqualsUtil {
28  
29      InmarsatCHistoryOceanRegion AOR_E = getOceanRegion(581, "EAST_ATLANTIC");
30      InmarsatCHistoryOceanRegion POR = getOceanRegion(582, "PACIFIC");
31      InmarsatCHistoryOceanRegion IOR = getOceanRegion(583, "INDIAN");
32      InmarsatCHistoryOceanRegion AOR_W = getOceanRegion(584, "WEST_ATLANTIC");
33  
34      @Test
35      public void testEqualsOceanRegionsBothEmpty() {
36          assertTrue("should be equal if both region sets are null", 
37                  EqualsUtil.equalsOceanRegions(null, null));
38          assertTrue("should be equal if both region sets are empty", 
39                  EqualsUtil.equalsOceanRegions(new HashSet<InmarsatCHistoryOceanRegion>(), new HashSet<InmarsatCHistoryOceanRegion>()));
40          assertTrue("should be equal if one is null and the other is empty",
41                  EqualsUtil.equalsOceanRegions(null, new HashSet<InmarsatCHistoryOceanRegion>()));
42          assertTrue("should be equal if one is null and the other is empty",
43                  EqualsUtil.equalsOceanRegions(new HashSet<InmarsatCHistoryOceanRegion>(), null));        
44      }
45  
46      @Test
47      public void testEqualsOceanRegionsOneEmpty() {
48          assertFalse("should not be equal if one is null and the other has regions",
49                  EqualsUtil.equalsOceanRegions(null, new HashSet<>(Arrays.asList(AOR_W))));
50          assertFalse("should not be equal if one is null and the other has regions",
51                  EqualsUtil.equalsOceanRegions(new HashSet<>(Arrays.asList(AOR_W)), null));
52      }
53  
54      @Test
55      public void testEqualsOceanRegionsSameSet() {
56          assertTrue("should be equal if ocean region sets are the same",
57                  EqualsUtil.equalsOceanRegions(new HashSet<>(Arrays.asList(AOR_W, POR)), new HashSet<>(Arrays.asList(POR, AOR_W))));
58      }
59  
60      @Test
61      public void testEqualsOceanRegionsDifferentSets() {
62          assertFalse("should not be equal if different sets of ocean regions",
63                  EqualsUtil.equalsOceanRegions(new HashSet<>(Arrays.asList(AOR_W, IOR)), new HashSet<>(Arrays.asList(AOR_W, POR))));
64      }
65  
66      private InmarsatCHistoryOceanRegion getOceanRegion(Integer code, String name) {
67          InmarsatCHistoryOceanRegion oceanRegion = new InmarsatCHistoryOceanRegion();
68          oceanRegion.setCode(code);
69          oceanRegion.setName(name);
70          return oceanRegion;
71      }
72  
73  }
