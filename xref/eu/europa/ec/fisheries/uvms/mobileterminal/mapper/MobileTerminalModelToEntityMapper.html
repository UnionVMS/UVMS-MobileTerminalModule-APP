45      public static MobileTerminal mapMobileTerminalEntity(MobileTerminal entity, MobileTerminalType model, String serialNumber, MobileTerminalPlugin plugin, String username, String comment, EventCodeEnum event) throws MobileTerminalModelMapperException {
46          if (model == null) throw new MobileTerminalModelMapperException("No mobile terminal to map");
47          if (entity.getGuid() == null) {
48              entity.setGuid(UUID.randomUUID().toString());
49          }
50          entity.setArchived(model.isArchived());
51          entity.setInactivated(model.isInactive());
52          entity.setPlugin(plugin);
53          entity.setSerialNo(serialNumber);
54  
55          try {
56              entity.setSource(getSourceTypeFromModel(model.getSource()));
57          } catch (EnumException e) {
58              LOG.error("[ Error when getting terminal source enum from source.] ", e.getMessage());
59              throw new MobileTerminalModelMapperException(e.getMessage());
60          }
61  
62          MobileTerminalTypeEnum type = MobileTerminalTypeEnum.getType(model.getType());
63          if (type == null) throw new MobileTerminalModelMapperException("Non valid mobile terminal type when mapping");
64          entity.setMobileTerminalType(type);
65          entity.setUpdateTime(DateUtils.getNowDateUTC());
66          entity.setUpdatedBy(username);
67  
68          try {
69              mapHistoryAttributes(entity, model, username, comment, event);
70          } catch (EnumException e) {
71              LOG.error("[ Error when mapping attribute types ]");
72              throw new MobileTerminalModelMapperException(e.getMessage());
73          }
74  
75          // Channels can only change for these events
76          if (event == EventCodeEnum.MODIFY || event == EventCodeEnum.CREATE) {
77              try {
78                  mapChannels(entity, model, username);
79              } catch (EnumException e) {
80                  LOG.error("[ Error when mapping channel field types ]");
81                  throw new MobileTerminalModelMapperException(e.getMessage());
82              }
83          }
84  
85          entity.setUpdateTime(DateUtils.getNowDateUTC());
86          entity.setUpdatedBy(username);
87  
88  
89          return entity;
90      }

92      public static MobileTerminal mapNewMobileTerminalEntity(MobileTerminalType model, String serialNumber, MobileTerminalPlugin plugin, String username) throws MobileTerminalModelMapperException {
93          if (model == null) throw new MobileTerminalModelMapperException("No mobile terminal to map");
94          return mapMobileTerminalEntity(new MobileTerminal(), model, serialNumber, plugin, username, MobileTerminalConstants.CREATE_COMMENT, EventCodeEnum.CREATE);
95      }

97      private static void mapChannels(MobileTerminal entity, MobileTerminalType model, String username) throws EnumException, MobileTerminalModelMapperException {
98          List<ComChannelType> modelChannels = model.getChannels();
99          Set<Channel> channels = new HashSet<>();
100         for (ComChannelType channelType : modelChannels) {
101             Channel channel = null;
102             if (entity != null) {
103                 for (Channel c : entity.getChannels()) {
104                     if (c.getGuid().equals(channelType.getGuid())) {
105                         channel = c;
106                         break;
107                     }
108                 }
109             }
110 
111             if (channel == null) {
112                 channel = new Channel();
113                 channel.setGuid(UUID.randomUUID().toString());
114             }
115             channel.setMobileTerminal(entity);
116             channel.setUpdateTime(DateUtils.getNowDateUTC());
117             channel.setUpdateUser(username);
118             channel.setArchived(false);
119 
120 
121             ChannelHistory history = new ChannelHistory();
122             history.setChannel(channel);
123             history.setUpdateTime(DateUtils.getNowDateUTC());
124             history.setName(channelType.getName());
125             history.setActive(true);
126             history.setUpdatedBy(username);
127             history.setMobileTerminalEvent(entity.getCurrentEvent());
128 
129             String attributes = mapComChannelAttributes(channelType.getAttributes());
130             history.setAttributes(attributes);
131             if (channel.getHistories().size() == 0) {
132                 history.setEventCodeType(EventCodeEnum.CREATE);
133             } else {
134                 history.setEventCodeType(EventCodeEnum.MODIFY);
135             }
136 
137             for (ComChannelCapability capability : channelType.getCapabilities()) {
138                 if (MobileTerminalConstants.CAPABILITY_CONFIGURABLE.equalsIgnoreCase(capability.getType()) && capability.isValue()) {
139                     entity.getCurrentEvent().setConfigChannel(channel);
140                     history.setConfigChannel(capability.isValue());
141                 }
142                 if (MobileTerminalConstants.CAPABILITY_DEFAULT_REPORTING.equalsIgnoreCase(capability.getType()) && capability.isValue()) {
143                     entity.getCurrentEvent().setDefaultChannel(channel);
144                     history.setDefaultChannel(capability.isValue());
145                 }
146                 if (MobileTerminalConstants.CAPABILITY_POLLABLE.equalsIgnoreCase(capability.getType()) && capability.isValue()) {
147                     entity.getCurrentEvent().setPollChannel(channel);
148                     history.setPollChannel(capability.isValue());
149                 }
150             }
151 
152             // No changes to channel means no new history
153             ChannelHistory channelHistory = channel.getCurrentHistory();
154             if (channelHistory != null) {
155                 if (channelHistory.equals(history)) {
156                     channels.add(channel);
157                     continue;
158                 } else {
159                     for (ChannelHistory ch : channel.getHistories()) {
160                         ch.setActive(false);
161                         channels.add(channel);
162                     }
163                 }
164             }
165 
166             for (ChannelHistory ch : channel.getHistories()) {
167                 ch.setActive(false);
168             }
169             channel.getHistories().add(history);
170             channels.add(channel);
171 
172         }
173 
174         for (Channel channel : entity.getChannels()) {
175             if (!channels.contains(channel)) {
176                 channel.setArchived(true);
177                 ChannelHistory current = channel.getCurrentHistory();
178                 current.setActive(false);
179 
180                 ChannelHistory archive = new ChannelHistory();
181                 archive.setEventCodeType(EventCodeEnum.ARCHIVE);
182                 archive.setName(current.getName());
183                 archive.setMobileTerminalEvent(entity.getCurrentEvent());
184                 archive.setActive(true);
185                 archive.setAttributes(current.getAttributes());
186                 archive.setChannel(channel);
187                 archive.setUpdatedBy(username);
188                 archive.setUpdateTime(DateUtils.getNowDateUTC());
189                 channel.getHistories().add(archive);
190             }
191         }
192 
193         entity.setChannels(channels);
194     }

196     private static void mapHistoryAttributes(MobileTerminal entity, MobileTerminalType model, String username, String comment, EventCodeEnum eventCode) throws EnumException, MobileTerminalModelMapperException {
197         List<MobileTerminalAttribute> modelAttributes = model.getAttributes();
198         MobileTerminalEvent current = entity.getCurrentEvent();
199 
200         MobileTerminalEvent history = new MobileTerminalEvent();
201         history.setActive(true);
202         history.setUpdateTime(DateUtils.getNowDateUTC());
203         history.setUpdatedBy(username);
204         history.setMobileTerminal(entity);
205         history.setComment(comment);
206         history.setEventCodeType(eventCode);
207         if (current != null && model.getConnectId() == null && eventCode != EventCodeEnum.UNLINK) {
208             history.setConnectId(current.getConnectId());
209         } else {
210             history.setConnectId(model.getConnectId());
211         }
212 
213         history.setAttributes(mapHistoryAttributes(modelAttributes));
214         for (MobileTerminalEvent event : entity.getMobileTerminalEvents()) {
215             event.setActive(false);
216         }
217         entity.getMobileTerminalEvents().add(history);
218     }

220     private static String mapHistoryAttributes(List<MobileTerminalAttribute> modelAttributes) {
221         StringBuffer sb = new StringBuffer();
222         for (MobileTerminalAttribute attr : modelAttributes) {
223             sb.append(attr.getType()).append("=");
224             sb.append(attr.getValue()).append(";");
225         }
226 
227         return sb.toString();
228     }
229 
230     private static String mapComChannelAttributes(List<ComChannelAttribute> modelAttributes) {
231         StringBuffer sb = new StringBuffer();
232         for (ComChannelAttribute attr : modelAttributes) {
233             sb.append(attr.getType()).append("=");
234             sb.append(attr.getValue()).append(";");
235         }
236 
237         return sb.toString();
238     }

240     private static MobileTerminalSourceEnum getSourceTypeFromModel(MobileTerminalSource model) throws EnumException {
241         if (model != null) {
242             switch (model) {
243                 case INTERNAL:
244                     return MobileTerminalSourceEnum.INTERNAL;
245                 case NATIONAL:
246                     return MobileTerminalSourceEnum.NATIONAL;
247             }
248         }
249         throw new EnumException("Couldn't map enum (from model) in " + MobileTerminalSourceEnum.class.getName());
250     }
