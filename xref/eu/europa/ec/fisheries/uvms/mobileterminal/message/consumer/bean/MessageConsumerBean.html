12  package eu.europa.ec.fisheries.uvms.mobileterminal.message.consumer.bean;
13  
14  import javax.ejb.ActivationConfigProperty;
15  import javax.ejb.MessageDriven;
16  import javax.ejb.TransactionAttribute;
17  import javax.ejb.TransactionAttributeType;
18  import javax.enterprise.event.Event;
19  import javax.inject.Inject;
20  import javax.jms.Message;
21  import javax.jms.MessageListener;
22  import javax.jms.TextMessage;
23  
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  import eu.europa.ec.fisheries.schema.mobileterminal.module.v1.MobileTerminalModuleBaseRequest;
28  import eu.europa.ec.fisheries.uvms.mobileterminal.message.constants.MessageConstants;
29  import eu.europa.ec.fisheries.uvms.mobileterminal.message.event.ErrorEvent;
30  import eu.europa.ec.fisheries.uvms.mobileterminal.message.event.GetReceivedEvent;
31  import eu.europa.ec.fisheries.uvms.mobileterminal.message.event.ListReceivedEvent;
32  import eu.europa.ec.fisheries.uvms.mobileterminal.message.event.PingReceivedEvent;
33  import eu.europa.ec.fisheries.uvms.mobileterminal.message.event.carrier.EventMessage;
34  import eu.europa.ec.fisheries.uvms.mobileterminal.model.exception.MobileTerminalUnmarshallException;
35  import eu.europa.ec.fisheries.uvms.mobileterminal.model.mapper.JAXBMarshaller;
36  
37  @MessageDriven(mappedName = MessageConstants.COMPONENT_EVENT_QUEUE, activationConfig = {
38          @ActivationConfigProperty(propertyName = "messagingType", propertyValue = MessageConstants.CONNECTION_TYPE),
39          @ActivationConfigProperty(propertyName = "destinationType", propertyValue = MessageConstants.DESTINATION_TYPE_QUEUE),
40          @ActivationConfigProperty(propertyName = "destination", propertyValue = MessageConstants.COMPONENT_EVENT_QUEUE_NAME),
41          @ActivationConfigProperty(propertyName = "destinationJndiName", propertyValue = MessageConstants.COMPONENT_EVENT_QUEUE),
42          @ActivationConfigProperty(propertyName = "connectionFactoryJndiName", propertyValue = MessageConstants.CONNECTION_FACTORY)
43  })
44  public class MessageConsumerBean implements MessageListener {
45  
46      final static Logger LOG = LoggerFactory.getLogger(MessageConsumerBean.class);
47  
48      @Inject
49      @PingReceivedEvent
50      Event<EventMessage> pingReceivedEvent;
51  
52      @Inject
53      @GetReceivedEvent
54      Event<EventMessage> getReceivedEvent;
55  
56      @Inject
57      @ListReceivedEvent
58      Event<EventMessage> listReceivedEvent;
59  
60  
61      @Inject
62      @ErrorEvent
63      Event<EventMessage> errorEvent;
64  
65      @Override
66      @TransactionAttribute(TransactionAttributeType.REQUIRES_NEW)
67      public void onMessage(Message message) {
68          
69          TextMessage textMessage = (TextMessage) message;
70          LOG.info("Message received in mobileterminal");
71          
72          try {
73              
74              MobileTerminalModuleBaseRequest request = JAXBMarshaller.unmarshallTextMessage(textMessage, MobileTerminalModuleBaseRequest.class);
75  
76              switch (request.getMethod()) {
77                  case GET_MOBILE_TERMINAL:
78                      getReceivedEvent.fire(new EventMessage(textMessage));
79                      break;
80                  case LIST_MOBILE_TERMINALS:
81                      listReceivedEvent.fire(new EventMessage(textMessage));
82                      break;
83                  case PING:
84                      pingReceivedEvent.fire(new EventMessage(textMessage));
85                      break;
86                  default:
87                      LOG.error("[ Unsupported request: {} ]", request.getMethod());
88                      break;
89              }
90  
91          } catch (NullPointerException | MobileTerminalUnmarshallException e) {
92              LOG.error("[ Error when receiving message in mobileterminal. ] {}", e.getMessage());
93              errorEvent.fire(new EventMessage(textMessage, "Error when receivning message in mobileterminal: " + e.getMessage()));
94          }
95      }
96  
97  }
