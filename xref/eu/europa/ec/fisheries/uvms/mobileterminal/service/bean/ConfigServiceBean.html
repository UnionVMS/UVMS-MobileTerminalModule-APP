42  @Stateless
43  public class ConfigServiceBean implements ConfigService {
44  
45      final static Logger LOG = LoggerFactory.getLogger(ConfigServiceBean.class);
46  
47      @EJB
48      MessageProducer messageProducer;
49  
50  	//@EJB(lookup = ServiceConstants.DB_ACCESS_CONFIG_MODEL)
51  	@EJB
52  	ConfigModel configModel;
53  
54      @EJB
55      MessageConsumer reciever;
56  
57      @Override
58      public List<TerminalSystemType> getTerminalSystems() throws MobileTerminalException {
59          LOG.debug("GET TERMINAL SYSTEM TRANSPONDERS INVOKED IN SERVICE LAYER");
60  		List<TerminalSystemType> systemList = configModel.getAllTerminalSystems();
61          return systemList;
62      }
63  
64  	@Override
65  	public List<ConfigList> getConfig() throws MobileTerminalException {
66  		LOG.debug("Get configuration in service layer");
67  		List<ConfigList> values = configModel.getConfigValues();
68  		return values;
69  	}
70  
71  	@Override
72  	public List<Plugin> upsertPlugins(List<PluginService> plugins, String username) throws MobileTerminalException {
73  		List<Plugin> plugin = configModel.upsertPlugins(plugins);
74  		return plugin;
75  	}
76  
77  	@Override
78  	public List<ServiceResponseType> getRegisteredMobileTerminalPlugins() {
79  		LOG.debug("Get registered service types");
80  		try {
81  			List<PluginType> pluginTypes = new ArrayList<>();
82  			pluginTypes.add(PluginType.SATELLITE_RECEIVER);
83  			String data = ExchangeModuleRequestMapper.createGetServiceListRequest(pluginTypes);
84  			String messageId = messageProducer.sendModuleMessage(data, ModuleQueue.EXCHANGE);
85  			TextMessage response = reciever.getMessage(messageId, TextMessage.class);
86  			return ExchangeModuleResponseMapper.mapServiceListResponse(response, messageId);
87  		} catch (ExchangeModelMapperException | MobileTerminalMessageException e) {
88  			LOG.error("Failed to map to exchange get service list request");
89  			return null;
90  		}
91  	}
92  }
