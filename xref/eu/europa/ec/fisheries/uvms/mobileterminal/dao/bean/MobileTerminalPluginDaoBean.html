12  package eu.europa.ec.fisheries.uvms.mobileterminal.dao.bean;
13  
14  import java.util.List;
15  
16  import javax.ejb.Stateless;
17  import javax.persistence.NoResultException;
18  import javax.persistence.TypedQuery;
19  
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  
23  import eu.europa.ec.fisheries.uvms.mobileterminal.constant.MobileTerminalConstants;
24  import eu.europa.ec.fisheries.uvms.mobileterminal.dao.Dao;
25  import eu.europa.ec.fisheries.uvms.mobileterminal.dao.MobileTerminalPluginDao;
26  import eu.europa.ec.fisheries.uvms.mobileterminal.dao.exception.ConfigDaoException;
27  import eu.europa.ec.fisheries.uvms.mobileterminal.dao.exception.NoEntityFoundException;
28  import eu.europa.ec.fisheries.uvms.mobileterminal.dao.exception.TerminalDaoException;
29  import eu.europa.ec.fisheries.uvms.mobileterminal.entity.MobileTerminalPlugin;
30  
31  @Stateless
32  public class MobileTerminalPluginDaoBean extends Dao implements MobileTerminalPluginDao {
33  	final static Logger LOG = LoggerFactory.getLogger(MobileTerminalPluginDaoBean.class);
34  
35  	@Override
36  	public List<MobileTerminalPlugin> getPluginList() throws ConfigDaoException {
37  		try {
38              TypedQuery<MobileTerminalPlugin> query = em.createNamedQuery(MobileTerminalConstants.PLUGIN_FIND_ALL, MobileTerminalPlugin.class);
39              return query.getResultList();
40          } catch (Exception e) {
41              LOG.error("[ Error when getting plugin list. ] " + e.getMessage());
42              throw new ConfigDaoException("No entities found when retrieving all plugins");
43          }
44  	}
45  	
46  	@Override
47  	public MobileTerminalPlugin createMobileTerminalPlugin(MobileTerminalPlugin plugin) throws TerminalDaoException {
48  		try {
49  			em.persist(plugin);
50  			return plugin;
51  		} catch (Exception e) {
52  			throw new TerminalDaoException("[ create mobile terminal plugin ] ");
53  		}
54  	}
55  
56  	@Override
57  	public MobileTerminalPlugin getPluginByServiceName(String serviceName) throws NoEntityFoundException {
58  		try {
59              TypedQuery<MobileTerminalPlugin> query = em.createNamedQuery(MobileTerminalConstants.PLUGIN_FIND_BY_SERVICE_NAME, MobileTerminalPlugin.class);
60              query.setParameter("serviceName", serviceName);
61              return query.getSingleResult();
62          } catch (NoResultException e) {
63              LOG.error("[ Error when getting plugin by service name. ] {}", e.getMessage());
64              throw new NoEntityFoundException("No entities found when retrieving mobile terminal plugin by service name");
65          }
66  	}
67  
68  	//ToDo: Look at if a null check is needed on an incoming MobileTerminalPlugin entity before allowing a merge.
69  	@Override
70  	public MobileTerminalPlugin updatePlugin(MobileTerminalPlugin entity) throws TerminalDaoException {
71  		try {
72  			em.merge(entity);
73  			em.flush();
74  			return entity;
75  		} catch (Exception e) {
76  			throw new TerminalDaoException("  [ update mobile terminal plugin ]");
77  		}
78  	}
79  }
